<script>
    const items = ["jeep", "Audi", "叉子", "橙球", "狗头", "炉子", "披萨", "球衣", "台灯"];
    const tools = [
        { id: 'h', name: 'Hunyuan' },
        { id: 't', name: 'Tripo' },
        { id: 'm', name: 'Meshy' },
        { id: 'Hy', name: 'Hyper3D' },
        { id: 'Tr', name: 'Trellis' }
    ];

    const container = document.getElementById('questions');

    items.forEach((item, index) => {
        // 使用 encodeURI 处理带空格的文件名路径
        const originSrc = encodeURI(`o ${item}.png`);
        
        let html = `
            <div class="card">
                <div class="card-title">Case ${index + 1}: ${item}</div>
                <div class="case-instruction">
                    请选最贴近参考图的模型，使用浏览器有放大功能！<br>
                    Please select the model that most closely matches the reference image. Your browser has a zoom function!
                </div>
                
                <div class="origin-section">
                    <img src="${originSrc}" class="origin-img" alt="Reference">
                    <div class="tag">参考原图 / REFERENCE IMAGE</div>
                </div>

                <input type="hidden" name="case_${item}" id="input_${item}" required>
                <div class="options-grid">
        `;

        const shuffled = [...tools].sort(() => Math.random() - 0.5);
        shuffled.forEach(tool => {
            const toolSrc = encodeURI(`${tool.id} ${item}.png`);
            html += `
                <div class="opt-item" onclick="select(this, '${item}', '${tool.name}')">
                    <img src="${toolSrc}" class="option-img" loading="lazy">
                </div>
            `;
        });
        html += `</div></div>`;
        container.innerHTML += html;
    });

    function select(el, itemKey, toolName) {
        const parent = el.parentElement;
        parent.querySelectorAll('.opt-item').forEach(i => i.classList.remove('active'));
        el.classList.add('active');
        document.getElementById('input_' + itemKey).value = toolName;
    }

    // 提交前的逻辑检查
    document.getElementById('surveyForm').onsubmit = function(e) {
        let unanswered = 0;
        items.forEach(item => {
            if (!document.getElementById('input_' + item).value) unanswered++;
        });
        
        if (unanswered > 0) {
            e.preventDefault();
            alert("您还有 " + unanswered + " 个选项未选择，请完成后提交！\nPlease complete all " + unanswered + " cases before submitting!");
        }
    };
</script>
